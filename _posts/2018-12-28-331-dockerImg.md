file system


app.py

```py
print("hello world")
```

Dockerfile

```shell
FROM ubuntu:15.04
COPY app.py /app
CMD python /app/app.py
```


```shell
docker build -t my_test_image .
docker save my_test_image > my_test_image
file my_test_image
mkdir unpacked_image
tar -xvf my_test_image -C unpacked_image
```
